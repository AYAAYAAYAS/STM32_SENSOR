python
import argparse
import time
import cv2
import numpy as np
import serial
import mediapipe as mp


# ================= 串口发送 =================
class SerialSender:
    def __init__(self, port, baud):
        self.port = port
        self.baud = baud
        self.ser = None

    def open(self):
        try:
            self.ser = serial.Serial(self.port, self.baud, timeout=0.1)
            time.sleep(0.05)
            print(f"串口已打开: {self.port} @ {self.baud}")
            return True
        except Exception as e:
            print(f"无法打开串口 {self.port}: {e}")
            self.ser = None
            return False

    def send(self, msg: str):
        if self.ser:
            try:
                self.ser.write(msg.encode())
            except Exception:
                pass

    def close(self):
        if self.ser:
            self.ser.close()


# ================= 参数 =================
def parse_args():
    p = argparse.ArgumentParser("灰度 + MediaPipe 手掌张开检测")
    p.add_argument("--port", default=None, help="串口号，如 COM3")
    p.add_argument("--baud", default=115200, type=int)
    p.add_argument("--camera", default=0, type=int)
    return p.parse_args()


# ================= 判断是否张开手掌 =================
def is_palm_open(hand_landmarks):
    """
    返回 True 表示手掌张开
    """
    lm = hand_landmarks.landmark
    fingers = []

    # 拇指（x 方向判断，假设右手）
    fingers.append(lm[4].x > lm[3].x)

    # 其余四指（y 方向）
    tips = [8, 12, 16, 20]
    pips = [6, 10, 14, 18]

    for tip, pip in zip(tips, pips):
        fingers.append(lm[tip].y < lm[pip].y)

    return sum(fingers) >= 4


# ================= 主程序 =================
def main():
    args = parse_args()

    sender = None
    if args.port:
        sender = SerialSender(args.port, args.baud)
        sender.open()

    cap = cv2.VideoCapture(args.camera)
    if not cap.isOpened():
        print("摄像头打开失败")
        return

    # MediaPipe Hands
    mp_hands = mp.solutions.hands
    mp_draw = mp.solutions.drawing_utils

    hands = mp_hands.Hands(
        static_image_mode=False,
        max_num_hands=1,
        min_detection_confidence=0.6,
        min_tracking_confidence=0.6
    )

    last_send_time = 0

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # 转灰度
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

        # 灰度 -> RGB（关键！）
        rgb = cv2.cvtColor(gray, cv2.COLOR_GRAY2RGB)

        # MediaPipe 处理
        results = hands.process(rgb)

        # 可视化
        vis = cv2.cvtColor(gray, cv2.COLOR_GRAY2BGR)

        if results.multi_hand_landmarks:
            for hand_landmarks in results.multi_hand_landmarks:
                mp_draw.draw_landmarks(
                    vis,
                    hand_landmarks,
                    mp_hands.HAND_CONNECTIONS
                )

                if is_palm_open(hand_landmarks):
                    cv2.putText(
                        vis, "PALM OPEN",
                        (30, 50),
                        cv2.FONT_HERSHEY_SIMPLEX,
                        1.2,
                        (0, 255, 0),
                        3
                    )

                    # 限速发送，避免串口刷爆
                    if time.time() - last_send_time > 0.5:
                        print("检测到张开手掌")
                        if sender:
                            sender.send("PALM_OPEN\n")
                        last_send_time = time.time()

        cv2.imshow("Gray + MediaPipe", vis)

        key = cv2.waitKey(1) & 0xFF
        if key == ord("q"):
            break

    cap.release()
    cv2.destroyAllWindows()
    if sender:
        sender.close()


if __name__ == "__main__":
    main()

python
import cv2
import time
import argparse
import serial
import numpy as np
from ultralytics import YOLO


# ================= 串口 =================
class SerialSender:
    def __init__(self, port, baud):
        self.port = port
        self.baud = baud
        self.ser = None

    def open(self):
        try:
            self.ser = serial.Serial(self.port, self.baud, timeout=0.1)
            time.sleep(0.05)
            print(f"串口已打开: {self.port}")
            return True
        except Exception as e:
            print("串口打开失败:", e)
            return False

    def send(self, msg):
        if self.ser:
            self.ser.write(msg.encode())

    def close(self):
        if self.ser:
            self.ser.close()


# ================= 参数 =================
def parse_args():
    p = argparse.ArgumentParser("YOLO 灰度手掌检测")
    p.add_argument("--camera", type=int, default=0)
    p.add_argument("--port", default=None)
    p.add_argument("--baud", type=int, default=115200)
    return p.parse_args()


# ================= 主程序 =================
def main():
    args = parse_args()

    sender = None
    if args.port:
        sender = SerialSender(args.port, args.baud)
        sender.open()

    cap = cv2.VideoCapture(args.camera)
    if not cap.isOpened():
        print("摄像头打开失败")
        return

    # YOLOv8 模型（通用目标检测，能检测人手）
    model = YOLO("yolov8n.pt")  # 轻量版，实时性好

    last_send = 0

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # 转灰度
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

        # YOLO 需要 3 通道，这里用“伪 RGB”
        rgb = cv2.cvtColor(gray, cv2.COLOR_GRAY2RGB)

        results = model(rgb, conf=0.4, verbose=False)

        vis = cv2.cvtColor(gray, cv2.COLOR_GRAY2BGR)

        for r in results:
            if r.boxes is None:
                continue

            for box in r.boxes:
                cls = int(box.cls[0])
                conf = float(box.conf[0])

                # COCO 中 person = 0（很多手在人体前）
                # 我们直接根据 bbox 形态判断“手掌”
                x1, y1, x2, y2 = map(int, box.xyxy[0])
                w = x2 - x1
                h = y2 - y1
                area = w * h
                ratio = w / (h + 1e-5)

                # 经验规则：张开手掌通常接近正方形
                if area > 5000 and 0.6 < ratio < 1.4:
                    cv2.rectangle(vis, (x1, y1), (x2, y2), (0, 255, 0), 2)
                    cv2.putText(vis, "PALM OPEN", (x1, y1 - 10),
                                cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 0), 2)

                    if time.time() - last_send > 0.5:
                        print("检测到张开手掌")
                        if sender:
                            sender.send("PALM_OPEN\n")
                        last_send = time.time()

        cv2.imshow("YOLO Palm Detection", vis)

        if cv2.waitKey(1) & 0xFF == ord("q"):
            break

    cap.release()
    cv2.destroyAllWindows()
    if sender:
        sender.close()


if __name__ == "__main__":
    main()
